class Aux{
     int ackermann(int m, int n) {
        if (m == 0) {
            return (n + 1);
        } else if (m > 0 && n == 0) {
            return ackermann(m - 1, 1);
        } else {
            return ackermann(m - 1, ackermann(m, n - 1));
        }
    }    
}
class Main{
    void main() {
        Aux eva = new Aux();
        println("3,8: "+eva.ackermann(3,8));    
    }  
}
